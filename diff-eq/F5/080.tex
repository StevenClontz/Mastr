\begin{exercise}
  \begin{exerciseTitle}F5: First order linear IVP (ver. 80)\end{exerciseTitle}
  \begin{exerciseStatement}
    
Find the solution to the given IVP.

    
\[4 \, t^{2} +ty'= 3 y \hspace{1em} y( -1 ) = 2\]

  \end{exerciseStatement}
  \begin{exerciseAnswer}
    
\[y= 2 t^ 3 +4 t^{2}\]

  \end{exerciseAnswer}
\end{exercise}
