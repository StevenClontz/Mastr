  \begin{exerciseTitle}F5: First order linear IVP (ver. 64)\end{exerciseTitle}
  \begin{exerciseStatement}
    
Find the solution to the given IVP.

    
\[-t^{2} +ty'= 3 y \hspace{1em} y( 1 ) = 1\]

  \end{exerciseStatement}
  \begin{exerciseAnswer}
    
\[y= 2 t^ 3 -1 t^{2}\]

  \end{exerciseAnswer}
