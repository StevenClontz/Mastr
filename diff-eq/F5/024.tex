  \begin{exerciseTitle}F5: First order linear IVP (ver. 24)\end{exerciseTitle}
  \begin{exerciseStatement}
    
Find the solution to the given IVP.

    
\[2 \, t^{3} +ty'= 2 y \hspace{1em} y( 2 ) = -32\]

  \end{exerciseStatement}
  \begin{exerciseAnswer}
    
\[y= -4 t^ 2 -2 t^{3}\]

  \end{exerciseAnswer}
