  \begin{exerciseStatement}
    
Find the solution to the given IVP.

    
\[-\frac{1}{t^{2}} +ty'= -3 y \hspace{1em} y( -1 ) = -2\]

  \end{exerciseStatement}
  \begin{exerciseAnswer}
    
\[y= 3 t^ -3 + \frac{1}{t^{2}}\]

  \end{exerciseAnswer}
