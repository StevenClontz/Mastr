\[\frac{10}{t^{2}} +ty'= 3 y \hspace{1em} y( 2 ) = \frac{33}{2}\]

  \end{exerciseStatement}

\[y= 2 t^ 3 +2 \frac{1}{t^{2}}\]

  \end{exerciseAnswer}
