\begin{exercise}
  \begin{exerciseTitle}C1 - Non-homogeneous first-order linear ODE (ver. 5)\end{exerciseTitle}
  \begin{exerciseStatement}
    
Find the general solution to the given ODE.

    
\[y'-1y= 2 \, e^{t}\]

  \end{exerciseStatement}
  \begin{exerciseAnswer}
    
\[y= k e^{t} + 2 \, t e^{t}\]

  \end{exerciseAnswer}
\end{exercise}
